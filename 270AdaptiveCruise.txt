			TCCR3B = 0;
			uint16_t time = TCNT3 *.5;			//convert dist and time
			uint16_t dist = TCNT3/148*2.54;
			//printf("time  is %u\n", time);
			//printf("dist  is %u\n", dist);
			if(dist < 36)
			{
				// set to cruise

				PORTC |= (1 << PORTC1);

				OCR1A = (F_CPU / 1000000) * 1500 - 1;
				TCCR1A = 0;
				TCCR1B = (1 << WGM12) | (1 << CS11); 
				TIMSK1 |= (1 << OCIE1A);
			}
			if(dist > 36)
			{
				// set to fast
				PORTC |= (1 << PORTC1);
				
				OCR1A = (F_CPU / 1000000) * 1700 - 1;
				TCCR1A = 0;
				TCCR1B = (1 << WGM12) | (1 << CS11); 
				TIMSK1 |= (1 << OCIE1A);
			}
			_delay_ms(500);
		}
    }
}
